            const projectSelect = document.getElementById('projectSelect');

            const getSpotlightBtn = document.getElementById('getSpotlightBtn');

            const spotlightOutput = document.getElementById('spotlightOutput');

            const loadingIndicator = document.getElementById('loadingIndicator');


            getSpotlightBtn.addEventListener('click', async () => {

                const selectedProject = projectSelect.value;

                if (!selectedProject) {

                    spotlightOutput.innerHTML = '<p class="text-red-500">Please select a project first.</p>';

                    return;

                }


                loadingIndicator.classList.remove('hidden');

                spotlightOutput.innerHTML = ''; // Clear previous content


                const prompt = `Provide a concise and engaging spotlight (around 100-150 words) on the Saudi Vision 2030 giga-project named ${selectedProject}. Focus on its unique features, primary goals, and how it contributes to the overall Vision 2030. Avoid generic statements and highlight specific, interesting aspects.`;


                let chatHistory = [];

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = "";

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    const result = await response.json();


                    if (result.candidates && result.candidates.length > 0 &&

                        result.candidates[0].content && result.candidates[0].content.parts &&

                        result.candidates[0].content.parts.length > 0) {

                        const text = result.candidates[0].content.parts[0].text;

                        spotlightOutput.innerHTML = `<p>${text}</p>`;

                    } else {

                        spotlightOutput.innerHTML = '<p class="text-red-500">Could not generate spotlight. Please try again.</p>';

                    }

                } catch (error) {

                    console.error('Error calling Gemini API:', error);

                    spotlightOutput.innerHTML = '<p class="text-red-500">An error occurred while fetching the spotlight. Please try again later.</p>';

                } finally {

                    loadingIndicator.classList.add('hidden');

                }

            });

        });
